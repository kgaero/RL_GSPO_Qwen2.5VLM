\begin{center}
{\LARGE \textbf{Checkpoint Scoring and Selection Flow}}\\[0.4em]
{\large Shared helpers, eval metrics, controller updates, score aliases, and report-level keep/discard}\\[0.6em]
\end{center}

\noindent\textbf{Key distinction.}
\emph{Keep / discard} is not a training-time controller verdict.
It is a \emph{report-layer recommendation} built mostly from alias labels plus one hard-coded recommended final checkpoint.

\vspace{0.6em}
\noindent\textbf{Flow A: runtime eval and controller loop}

\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c c c c c}
\flowbox{Model completions\\ + gold answers} &
$\rightarrow$ &
\flowbox{Shared parsing helpers\\ float parse\\ exact-match check\\ solution-tag check\\ reasoning-tag check\\ tolerance check} &
$\rightarrow$ &
\flowbox{Eval metrics\\ exact match\\ tolerance accuracy\\ parseable rate\\ tag compliance\\ malformed rate\\ truncation rate} \\
& & $\searrow$ & & $\downarrow$ \\
& & & &
\flowbox{Reward controller\\ metric-gated weight updates\\ parseability guard\\ formatting guard\\ finish guard\\ correctness escalation} \\
\end{tabular}
\end{center}

\vspace{0.3em}
\noindent\textbf{Flow B: checkpoint scoring, aliases, and report recommendation}

\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c c c c c}
\flowbox{Eval metrics} &
$\rightarrow$ &
\flowbox{Weighted checkpoint scores\\ \code{structure\_score}\\ \code{correctness\_score}\\ \code{composite\_score}} &
$\rightarrow$ &
\flowbox{Run-local aliases\\ \code{latest}\\ \code{best\_structure}\\ \code{best\_correctness}\\ \code{best\_composite}} \\
& & & & $\downarrow$ \\
& & & & \flowbox{Report-level\\ \code{Keep / Discard}\\ recommendation} \\
\end{tabular}
\end{center}
